# 推荐用 nvm 安装 Node，避免 npm -g 权限问题
curl -fsSL https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash
source ~/.nvm/nvm.sh
nvm install --lts

# 验证
node -v
npm -v
git --version
